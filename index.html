Setup

<link rel="stylesheet" href="./dist/css/oblurlay.css">
<script src="//ajax.googleapis.com/ajax/libs/jquery/1.11.0/jquery.min.js"></script>
<script src="./dist/js/oblurlay.js"></script>
<script>
 $(document).ready(function() {
   $('.oblurlay').oblurlay();
 });  
</script>

<div class="oblurlay">
  <div class="oblurlay-upper">   
    upper layer
  </div>
  <div class="oblurlay-contents">
    contents layer
  </div>
</div>
